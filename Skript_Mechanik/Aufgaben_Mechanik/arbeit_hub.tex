					\RI{E}{pot} = m\cdot g\cdot h= \SI{5}{kg}\cdot\SI{9.81}{m/s^2}\SI{3}{m}=\SI{147.15}{J}
				\end{eqnarray*}
		\end{itemize}
	\end{loesung}
